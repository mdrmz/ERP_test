-- 1. MÜŞTERİLER TABLOSU GÜNCELLEMESİ (CRM Entegrasyonu)
-- Yeni sütunlar eklendi ve tablo yapısı modernize edildi.
DROP TABLE IF EXISTS musteriler;
CREATE TABLE musteriler (
    id INT(11) AUTO_INCREMENT PRIMARY KEY,
    cari_kod VARCHAR(50) NOT NULL UNIQUE,
    cari_tip ENUM('Müşteri', 'Tedarikçi') NOT NULL,
    firma_adi VARCHAR(255) NOT NULL,
    yetkili_kisi VARCHAR(100),
    telefon VARCHAR(50),
    eposta VARCHAR(100),
    vergi_dairesi VARCHAR(100),
    vergi_no VARCHAR(50),
    il VARCHAR(50),
    ilce VARCHAR(50),
    adres TEXT,
    ozel_notlar TEXT,
    bakiye DECIMAL(18,2) DEFAULT 0.00,
    para_birimi VARCHAR(10) DEFAULT 'TL',
    kayit_tarihi TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;

-- 2. SİPARİŞLER TABLOSU EKLEMELERİ
-- Sevkiyat ve ödeme takibi için yeni alanlar.
ALTER TABLE siparisler ADD COLUMN alici_adi VARCHAR(255) NULL AFTER musteri_id;
ALTER TABLE siparisler ADD COLUMN odeme_tarihi DATE NULL AFTER teslim_tarihi;

-- 3. BAKIM VE LABORATUVAR MODÜLÜ (Yeni Tablo)
-- Laboratuvar kimyasalları ve malzemeleri takibi için.
CREATE TABLE IF NOT EXISTS bakim_lab_malzemeler (
    id INT AUTO_INCREMENT PRIMARY KEY,
    malzeme_adi VARCHAR(255) NOT NULL,
    miktar DECIMAL(10,2) DEFAULT 0.00,
    birim VARCHAR(50),
    kullanim_alani VARCHAR(255),
    kritik_seviye DECIMAL(10,2) DEFAULT 0.00,
    tarih TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4;
